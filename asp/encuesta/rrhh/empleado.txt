Function Nz(v, alt)
    If IsNull(v) Or IsEmpty(v) Or Trim(CStr(v & "")) = "" Then
        Nz = alt
    Else
        Nz = v
    End If
End Function

Function SqlSafe(v)
    SqlSafe = Replace(Trim(CStr(v & "")), "'", "''")
End Function

Function EstadoClase(estado)
    Dim s
    s = UCase(Trim(CStr(estado & "")))

    Select Case s
        Case "OK", "PROCESADO", "DISPONIBLE", "NOTIFICADO", "LEIDO", "FIRMADO"
            EstadoClase = "ok"
        Case "ERROR"
            EstadoClase = "err"
        Case "PENDIENTE", "PROCESANDO"
            EstadoClase = "pen"
        Case "DUPLICADO"
            EstadoClase = "dup"
        Case Else
            EstadoClase = "neu"
    End Select
End Function

Dim empleadoID, reciboid, accion, observacionFirma
Dim msgOk, msgErr

accion           = LCase(Trim(Request("accion")))
observacionFirma = Trim(Request("observacionfirma"))

msgOk  = ""
msgErr = ""

If reciboid <= 0 Then
    Response.Write "ReciboID inválido."
    Response.End
End If

If accion = "leer" Then
    conn.Execute "EXEC empleado.usp_Empleado_Recibo_MarcarLeido " & empleadoID & ", " & reciboid
    msgOk = "Recibo marcado como leído."
End If

If accion = "firmar" Then
    conn.Execute "EXEC empleado.usp_Empleado_Recibo_Firmar " & _
                 "@EmpleadoID=" & empleadoID & ", " & _
                 "@ReciboID=" & reciboid & ", " & _
                 "@ObservacionFirma='" & SqlSafe(observacionFirma) & "'"
    msgOk = "Recibo firmado correctamente."
End If

Dim rs
Set rs = conn.Execute("EXEC empleado.usp_Empleado_Recibo_Detalle_Sel " & empleadoID & ", " & reciboid)

If rs.EOF Then
    Response.Write "No existe el recibo indicado o no pertenece al empleado."
    Response.End
End If
